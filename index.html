Creating a table


Create Table Table_Name (
		Column 1 data type and constrain,
			Column 2 data type,
			.
			.
			.
			Column n datatype,
			Primary key (Column 1)
					);

Example: 
Create Table Student (
			ID int, 
			Name varchar (50),
			Address varchar (100),
			City (15),
			Primary key (ID)
				);

Dropping a table


Drop Table Table_Name;

Example: 
Drop Table Student;

Altering a table


For adding a new column:
Alter Table Table_Name
	Add Column_Name Data type;
	
Example: 
Alter Table Student
	Add Marks int;


For Deleting a column:
Alter Table Table_Name
	Drop Column_Name;
	
Example: 
Alter Table Student
	Drop Address;


For Renaming a column:	
Alter Table Table_Name 
	Rename Old_Column_Name to New_Column_Name;
	
Example: 
Alter Table Table_Name 
	Rename ID to Student_ID;


For Renaming a table:	
Alter Table Table_Name 
	Rename New_Table_Name;
	
Example: 
Alter Table Table_Name 
	Rename Student_Info;

Truncating a table


Truncate Table Table_Name;
                                             
Example: 
Truncate Table Student;
